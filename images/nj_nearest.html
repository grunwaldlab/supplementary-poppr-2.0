## An exploration of collapsing MLGs using Nearest Neighbor
library(animation)
library(phangorn)
library(ape)
library(poppr)
library(adegenet)
library(ade4)
for (i in unique(fstats[[measure]]$thresholds)) {
    z <- mlg.filter(x, threshold = i, missing = "ignore", 
        algorithm = measure, distance = distfun, stats = "MLGs", 
        ...)
    if (isnj) {
        color_mlg_tree(x, the_tree, z, type = "u", lab4ut = "axial")
    }
    else {
        color_mlg_tree(x, the_tree, z)
    }
    if (isnj) {
        add.scale.bar()
    }
    else {
        axisPhylo(1)
    }
    title(paste("Threshold:", round(i, 3), "MLG:", length(unique(z))))
}
